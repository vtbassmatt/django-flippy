 7class BaseBackend(metaclass=ABCMeta):
 8    """
 9    Flipper established a protocol for adapters to implement.
10    This is the Pythonic implementation of it.
11    
12    https://www.flippercloud.io/docs/adapters/new
13    """
14    @abstractmethod
15    def features(self) -> set[FeatureName]:
16        "Get the set of known features."
17        pass
18
19    @abstractmethod
20    def add(self, feature: FeatureName) -> bool:
21        "Add a feature to the set of known features."
22        pass
23
24    @abstractmethod
25    def remove(self, feature: FeatureName) -> bool:
26        "Remove a feature from the set of known features."
27        pass
28
29    @abstractmethod
30    def clear(self, feature: FeatureName) -> bool:
31        "Clear all gate values for a feature."
32        pass
33
34    @abstractmethod
35    def get(self, feature: FeatureName) -> Feature:
36        "Get all gate values for a feature."
37        pass
38
39    @abstractmethod
40    def enable(self, feature: FeatureName, gate: Gate, thing: str | int | None=None) -> bool:
41        "Enable a gate for a thing."
42        pass
43
44    @abstractmethod
45    def disable(self, feature: FeatureName, gate: Gate, thing: str | int | None=None) -> bool:
46        "Disable a gate for a thing."
47        pass
48
49    @abstractmethod
50    def get_multi(self, features: list[FeatureName]) -> list[Feature]:
51        "Get all gate values for several features at once."
52        # default implementation; feel free to use or override
53        values = []
54        for feature in features:
55            values.append(self.get(feature))
56        return values
57
58    @abstractmethod
59    def get_all(self) -> list[Feature]:
60        "Get all gate values for all features at once."
61        # default implementation; feel free to use or override
62        values = []
63        for feature in self.features():
64            values.append(self.get(feature))
65        return values
66
67    # dict implementation
68    # note, there's no setting of values because that doesn't fit the
69    # dict contract cleanly
70    def __len__(self):
71        return len(self.features())
72    
73    def __getitem__(self, key: FeatureName):
74        return self.get(key)
75    
76    def __delitem__(self, key: FeatureName):
77        return self.remove(key)
78    
79    def __iter__(self):
80        return iter(self.features())
81    
82    def __contains__(self, key: FeatureName):
83        return key in self.features()
84    # end dict implementation

 24    def add(self, feature: FeatureName) -> bool:
 25        "Add a feature to the set of known features."
 26        try:

 28            return True
 29        except IntegrityError:
 30            return False
 31
 32    def remove(self, feature: FeatureName) -> bool:
 33        "Remove a feature from the set of known features."
 34        try:
 35            feature = FlippyFeature.objects.get(key=feature)
 36            feature.delete()
 37            return True
 38        except FlippyFeature.DoesNotExist:
 39            return False
 40
 41    def clear(self, feature: FeatureName) -> bool:
 42        "Clear all gate values for a feature."
 43        try:
 44            feature = FlippyFeature.objects.get(key=feature)
 45            feature.clear()
 46            return True
 47        except FlippyFeature.DoesNotExist:
 48            return False
 49
 50    def get(self, feature: FeatureName) -> Feature:
 51        "Get all gate values for a feature."
 52        try:
 53            feature = FlippyFeature.objects.get(key=feature)
 54            return feature.as_feature()
 55        except FlippyFeature.DoesNotExist:
 56            raise FeatureNotFound(feature)
 57
 58    def enable(self, feature: FeatureName, gate: Gate, thing: str | int | None=None) -> bool:
 59        "Enable a gate for a thing."
 60        try:
 61            feature = FlippyFeature.objects.get(key=feature)
 62        except FlippyFeature.DoesNotExist:
 63            return False
 64
 65        match gate:
 66            case Gate.Boolean:
 67                feature.boolean = True
 68                feature.save()
 69                return True
 70            case Gate.Actors:
 71                try:
 72                    feature.enabled_actors.create(key=thing)
 73                    return True
 74                except IntegrityError:
 75                    return False
 76            case Gate.Groups:
 77                try:
 78                    feature.enabled_groups.create(key=thing)
 79                    return True
 80                except IntegrityError:
 81                    return False
 82            case Gate.PercentageOfActors:
 83                try:
 84                    feature.percentage_of_actors = thing
 85                    feature.save()
 86                    return True
 87                except ValidationError:
 88                    return False
 89            case Gate.PercentageOfTime:
 90                try:
 91                    feature.percentage_of_time = thing
 92                    feature.save()
 93                    return True
 94                except ValidationError:
 95                    return False
 96            case Gate.Expression:
 97                raise NotImplementedError("ExpressionGate isn't supported")
 98            case _:
 99                raise ValueError(f"{gate} is not a known gate type")
100

102        "Disable a gate for a thing."

class FlipperCloudBackend(flippy.backends.BaseBackend): 
 36class FlipperCloudBackend(BaseBackend):
 37    """
 38    Get flags direct from Flipper Cloud.
 39    
 40    Note that this is not the preferred implementation, as it'll cause a service
 41    to service call on every invocation. However, it's good for scaffolding and
 42    kicking tires.
 43    """
 44    def __init__(self, token: str):
 45        if not token:
 46            raise ValueError('must pass a Flipper Cloud token')
 47
 48        self.client = httpx.Client(
 49            base_url=FLIPPER_CLOUD_BASE_URL,
 50            headers=HEADERS | { "Flipper-Cloud-Token": token },
 51        )
 52
 53    def features(self) -> set[FeatureName]:
 54        "Get the set of known features."
 55        r = self.client.get('/features?exclude_gate_names=true')
 56        self._raise_from_cloud(r)
 57
 58        return {f['key'] for f in r.json()['features']}
 59
 60    def add(self, feature: FeatureName) -> bool:
 61        "Add a feature to the set of known features."
 62        body = { 'name': feature }
 63        r = self.client.post('/features', json=body)
 64        self._raise_from_cloud(r, { 'feature': feature })
 65        return r.is_success
 66
 67    def remove(self, feature: FeatureName) -> bool:
 68        "Remove a feature from the set of known features."
 69        r = self.client.delete(f'/features/{feature}')
 70        self._raise_from_cloud(r, { 'feature': feature })
 71        return r.is_success
 72
 73    def clear(self, feature: FeatureName) -> bool:
 74        "Clear all gate values for a feature."
 75        r = self.client.delete(f'features/{feature}/clear')
 76        self._raise_from_cloud(r, { 'feature': feature })
 77        return r.is_success
 78
 79    def get(self, feature: FeatureName) -> Feature:
 80        "Get all gate values for a feature."
 81        r = self.client.get(f'/features/{feature}')
 82
 83        self._raise_from_cloud(r, { 'feature': feature })
 84        return Feature.from_api(r.json())        
 85
 86    def enable(self, feature: FeatureName, gate: Gate, thing: str | int | None=None) -> bool:
 87        "Enable a gate for a thing."
 88        body = self._body_for_gate(gate, thing)
 89        qs = { 'allow_unregistered_groups': 'true' } if gate == Gate.Groups else {}
 90        r = self.client.post(
 91            f'/features/{feature}/{gate.value}',
 92            json=body,
 93            params=qs,
 94        )
 95        
 96        self._raise_from_cloud(r, { 'feature': feature })
 97        return r.is_success
 98
 99    def disable(self, feature: FeatureName, gate: Gate, thing: str | int | None=None) -> bool:
100        "Disable a gate for a thing."
101        body = self._body_for_gate(gate, thing)
102        qs = { 'allow_unregistered_groups': 'true' } if gate == Gate.Groups else {}
103        # httpx doesn't accept `json` on .delete(), but the body is a required part
104        # of some disable requests
105        req = self.client.build_request(
106            'DELETE',
107            f'/features/{feature}/{gate.value}',
108            json=body,
109            params=qs,
110        )
111        r = self.client.send(req)
112        self._raise_from_cloud(r, { 'feature': feature })
113
114        return r.is_success
115
116    def get_multi(self, features: list[FeatureName]) -> list[Feature]:
117        "Get all gate values for several features at once."
118        qs = {
119            'exclude_gate_names': 'true',
120            'keys': ','.join(features),
121        }
122        r = self.client.get('/features', params=qs)
123        self._raise_from_cloud(r)
124
125        return [Feature.from_api(f) for f in r.json()['features']]
126
127    def get_all(self) -> list[Feature]:
128        "Get all gate values for all features at once."
129        qs = {
130            'exclude_gate_names': 'true',
131        }
132        r = self.client.get('/features', params=qs)
133        self._raise_from_cloud(r)
134
135        return [Feature.from_api(f) for f in r.json()['features']]
136
137    def _body_for_gate(self, gate: Gate, thing: str | int | None) -> dict:
138        match gate:
139            case Gate.Boolean:
140                return {}
141            case Gate.Actors:
142                return { 'flipper_id': str(thing) }
143            case Gate.Groups:
144                return { 'name': str(thing) }
145            case Gate.PercentageOfActors | Gate.PercentageOfTime:
146                return { 'percentage': str(thing) }
147            case Gate.Expression:
148                raise NotImplementedError("ExpressionGate isn't supported")
149            case _:
150                raise ValueError(f"{gate} is not a known gate type")
151
152    def _raise_from_cloud(self, response: httpx.Response, extra: dict = {}):
153        # https://www.flippercloud.io/docs/api#error-code-reference
154        if response.is_client_error or response.is_success:
155            try:
156                result = response.json()
157            except JSONDecodeError:
158                result = {}
159            if 'code' not in result:
160                return
161            
162            match result['code']:
163                case 1:
164                    raise FeatureNotFound(extra.get('feature', ''))
165                case 2:
166                    raise GroupNotRegistered(result.get('message', ''))
167                case 3:
168                    raise PercentageInvalid(result.get('message', ''))
169                case 4:
170                    raise FlipperIdInvalid(result.get('message', ''))
171                case 5:
172                    raise NameInvalid(result.get('message', ''))
173
174        response.raise_for_status()
